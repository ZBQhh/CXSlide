\section{设计原则}

\begin{frame}{内容与样式分离}
  \cxTimeEstimate{2}
  \begin{itemize}[<+->]
    \item 内容作者只表达语义，不直接指定颜色、字号和间距。
    \item 强调统一通过 \alert{语义命令} 和区块环境完成。
    \item 极端场景下仍保持一致的视觉规则和可读性。
    \item 同一份内容可以切换调色板、页脚、输出模式，而无需改正文案。
  \end{itemize}
\end{frame}

\begin{frame}{列表、编号与区块}
  \cxTwoCols[0.48]{%
    \begin{itemize}[<+->]
      \item 一级要点使用统一标记。
      \item 支持嵌套列表。
      \begin{itemize}
        \item 子项 A
        \item 子项 B
      \end{itemize}
      \item 支持 alert 与 example 语义。
    \end{itemize}
  }[0.48]{%
    \begin{enumerate}
      \item 问题定义
      \item 方案设计
      \item 风险控制
    \end{enumerate}
  }
\end{frame}

\begin{frame}{区块语义}
  \begin{block}{定理}
    $e^{i\pi}+1=0$
  \end{block}
  \begin{alertblock}{注意}
    此公式联结了五个基本常数。
  \end{alertblock}
  \begin{exampleblock}{例}
    取 $\theta=\pi$，代入 $e^{i\theta}=\cos\theta+i\sin\theta$。
  \end{exampleblock}
\end{frame}

\section{数学与排版}

\begin{frame}{公式高亮与注释}
  Fourier 变换：
  \begin{align}
    \hat{f}(\xi)
      &= \int_{-\infty}^{\infty}
           f(x)\, e^{-2\pi i x \xi}\, dx \\
    \uncover<2->{
      \widehat{f*g}(\xi)
      &= \hlmath<2>{\hat{f}(\xi)\cdot\hat{g}(\xi)}
    }
  \end{align}
  \onslide<2->{\eqnote{卷积定理}}
\end{frame}

\begin{frame}{密集公式}
  \begin{smallmath}
    \begin{align}
      \nabla\cdot\mathbf{E}
        &= \frac{\rho}{\varepsilon_0} \\
      \nabla\cdot\mathbf{B}
        &= 0 \\
      \nabla\times\mathbf{E}
        &= -\frac{\partial\mathbf{B}}{\partial t} \\
      \nabla\times\mathbf{B}
        &= \mu_0\mathbf{J}
          +\mu_0\varepsilon_0
           \frac{\partial\mathbf{E}}{\partial t}
    \end{align}
  \end{smallmath}
\end{frame}

\begin{frame}{数学盒与极端压缩}
  \begin{tinymath}
    \begin{align}
      \mathbox<1>{\sum_{k=1}^{n} k}
      &= \mathbox<1>{\frac{n(n+1)}{2}} \\
      \mathbox<2>{\sum_{k=1}^{n} k^2}
      &= \mathbox<2>{\frac{n(n+1)(2n+1)}{6}}
    \end{align}
  \end{tinymath}

  \cxSummary{排版策略}{%
    在信息密度上升时，优先压缩留白，再降级到局部数学字号，而不是整体缩小整页。%
  }
\end{frame}

\begin{frame}[fragile]{代码排版}
\begin{lstlisting}[language=Python]
def fibonacci(n):
    """Calculate the n-th Fibonacci number."""
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)
\end{lstlisting}
  内联代码：\code{fibonacci(10) = 55}
\end{frame}
