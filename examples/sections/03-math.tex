\section{数学与排版}

\begin{frame}{行内与陈列公式}
  行内公式示例：$A = \{x \in \mathbb{R} : x > 0\}$。

  \begin{equation}
    \int_{0}^{1} x^2 \, \mathrm{d}x = \frac{1}{3}
  \end{equation}
\end{frame}

\begin{frame}{高亮与紧凑数学}
  重点结论可以写成
  \[
    \hlmath{f(x) = \sum_{n=0}^{\infty} a_n x^n}
  \]

  紧凑场景下也可以直接书写简洁公式：
  \[
    \nabla \cdot \mathbf{E} = \frac{\rho}{\varepsilon_0}
  \]
\end{frame}

\begin{frame}{定理型内容}
  \begin{block}{柯西不等式}
    \[
      |\langle \mathbf{u}, \mathbf{v} \rangle|^2
      \le \langle \mathbf{u}, \mathbf{u} \rangle
      \langle \mathbf{v}, \mathbf{v} \rangle
    \]
  \end{block}
  \begin{exampleblock}{解释}
    等号成立当且仅当两个向量线性相关。
  \end{exampleblock}
\end{frame}
