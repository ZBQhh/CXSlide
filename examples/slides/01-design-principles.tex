\section{设计原则}

\begin{frame}{内容与样式分离}
  \cxTimeEstimate{2}
  \begin{itemize}[<+->]
    \item 内容作者只表达语义，不直接指定颜色、字号和间距。
    \item 强调统一通过 \alert{语义命令} 和区块环境完成。
    \item 极端场景下仍保持一致的视觉规则和可读性。
    \item 同一份内容可以切换调色板、页脚、输出模式，而无需改正文案。
  \end{itemize}
\end{frame}

\begin{frame}{列表、编号与区块}
  \cxTwoCols[0.48]{%
    \begin{itemize}[<+->]
      \item 一级要点使用统一标记。
      \item 支持嵌套列表。
      \begin{itemize}
        \item 子项 A
        \item 子项 B
      \end{itemize}
      \item 支持 alert 与 example 语义。
    \end{itemize}
  }[0.48]{%
    \begin{enumerate}
      \item 问题定义
      \item 方案设计
      \item 风险控制
    \end{enumerate}
  }
\end{frame}

\begin{frame}{区块语义}
  \begin{block}{定理}
    $e^{i\pi}+1=0$
  \end{block}
  \begin{alertblock}{注意}
    此公式联结了五个基本常数。
  \end{alertblock}
  \begin{exampleblock}{例}
    取 $\theta=\pi$，代入 $e^{i\theta}=\cos\theta+i\sin\theta$。
  \end{exampleblock}
\end{frame}
